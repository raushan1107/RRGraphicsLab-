    Your First OpenGL Window
    Module 2 · GLFW + GLEW Setup · Navy dark background
    Beginner ›
    ℹ
    What this demo does: Opens an 800×600 window with a dark navy background (R=0.05, G=0.10, B=0.20). Prints your GPU name, OpenGL version, and GLSL version. Press ESC to close. This is the foundation that every OpenGL application starts with.

    💻 main.cpp — Complete Code

    C++
    // Demo 1 — First OpenGL Window | RR Graphics Lab
// Instructor: Raushan Ranjan, MCT
#include <GL/glew.h>   // ALWAYS before GLFW
#include <GLFW/glfw3.h>
#include <iostream>

// Called automatically when window is resized
void framebuffer_size_callback(GLFWwindow* w, int width, int h){
    glViewport(0,0,width,h); // update drawing area
}

void process_input(GLFWwindow* w){
    // ESC key closes the window
    if(glfwGetKey(w, GLFW_KEY_ESCAPE)==GLFW_PRESS)
        glfwSetWindowShouldClose(w, true);
}

int main(){
    glfwInit();
    // Request OpenGL 3.3 Core Profile (modern, no legacy)
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    GLFWwindow* win = glfwCreateWindow(800,600,
        "RR Graphics Lab — Module 2", NULL, NULL);
    glfwMakeContextCurrent(win); // make this window's context active
    glfwSetFramebufferSizeCallback(win, framebuffer_size_callback);

    // GLEW loads all modern OpenGL functions from the driver
    glewExperimental = GL_TRUE;
    glewInit();

    // Print GPU info — always useful for debugging
    std::cout << "GPU: "     << glGetString(GL_RENDERER) << "\n";
    std::cout << "OpenGL: "  << glGetString(GL_VERSION)  << "\n";
    std::cout << "GLSL: "    << glGetString(GL_SHADING_LANGUAGE_VERSION) << "\n";

    glViewport(0, 0, 800, 600);

    // ── THE RENDER LOOP — runs 60+ times per second ──
    while(!glfwWindowShouldClose(win)){
        process_input(win);

        // Navy blue background — RGB all in 0.0–1.0 range
        glClearColor(0.05f, 0.10f, 0.20f, 1.0f);
        glClear(GL_COLOR_BUFFER_BIT);

        glfwSwapBuffers(win); // show the back buffer
        glfwPollEvents();     // handle OS events
    }

    glfwDestroyWindow(win);
    glfwTerminate();
}

    ✓

    🧪 Experiments — Try These

    1. Change background color:

      Try glClearColor(1.0f, 0.0f, 0.0f, 1.0f) for red. Values are 0.0–1.0 (not 0–255). To convert: divide RGB by 255.

    2. Comment out glfwPollEvents():

      The window freezes — you can't move or close it. This proves why event polling is mandatory every frame.

    3. Comment out glfwSwapBuffers():

      Window stays black even though you set a clear color. Proves double-buffering — you draw to the back buffer, but never show it.

    🚢
    Naval Display System
    Every display system — radar, sonar, navigation — starts with exactly this: a context, a window, and a render loop refreshing at 60Hz.
    ✈️
    Flight Simulation
    The render loop timing directly impacts simulator fidelity. A 60fps loop gives 16.7ms per frame — the budget your entire scene must fit within.

    First Triangle with VBO & VAO
    Module 3 · GPU Buffers · Vertex Shaders
    Beginner ›
    ℹ
    What this demo does: Renders a solid orange triangle using a VBO (vertex data on GPU), VAO (remembers the data format), and two minimal shaders. This is the "Hello World" of GPU rendering — every 3D scene ever made uses this exact pattern.
    ⚠
    VBO vs VAO mental model: The VBO is the filing cabinet (raw data). The VAO is the instruction manual that says "read 3 floats at a time, starting at offset 0." The VAO remembers the format so you don't re-configure it every frame.

    💻 Complete Code

    C++
    // Demo 2 — First Triangle | RR Graphics Lab
#include <GL/glew.h>
#include <GLFW/glfw3.h>
#include <iostream>

// Vertex shader: runs once per vertex
// layout(location=0) links to VAO attribute index 0
const char* vertSrc = R"(
#version 330 core
layout(location=0) in vec3 aPos;
void main(){ gl_Position = vec4(aPos,1.0); }
)";

// Fragment shader: runs once per pixel — outputs orange
const char* fragSrc = R"(
#version 330 core
out vec4 FragColor;
void main(){ FragColor = vec4(1.0,0.5,0.2,1.0); }
)";

unsigned int compileShader(unsigned int type, const char* src){
    unsigned int id = glCreateShader(type);
    glShaderSource(id, 1, &src, NULL);
    glCompileShader(id);
    int ok; char log[512];
    glGetShaderiv(id, GL_COMPILE_STATUS, &ok);
    if(!ok){ glGetShaderInfoLog(id,512,NULL,log);
              std::cerr << "Shader error: " << log; }
    return id;
}

int main(){
    glfwInit();
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR,3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR,3);
    glfwWindowHint(GLFW_OPENGL_PROFILE,GLFW_OPENGL_CORE_PROFILE);
    GLFWwindow* win = glfwCreateWindow(800,600,"Triangle Demo",NULL,NULL);
    glfwMakeContextCurrent(win);
    glewExperimental=GL_TRUE; glewInit();

    // Compile and link shaders into a program
    unsigned int vs = compileShader(GL_VERTEX_SHADER,   vertSrc);
    unsigned int fs = compileShader(GL_FRAGMENT_SHADER, fragSrc);
    unsigned int prog = glCreateProgram();
    glAttachShader(prog,vs); glAttachShader(prog,fs);
    glLinkProgram(prog);
    glDeleteShader(vs); glDeleteShader(fs);

    // Triangle vertices in NDC (-1 to 1 range)
    float verts[] = {
        -0.5f, -0.5f, 0.0f,  // bottom-left
         0.5f, -0.5f, 0.0f,  // bottom-right
         0.0f,  0.5f, 0.0f   // top-center
    };

    unsigned int VAO, VBO;
    glGenVertexArrays(1, &VAO);  // create VAO (remembers format)
    glGenBuffers(1, &VBO);        // create VBO (holds data)

    glBindVertexArray(VAO);       // start recording attribute config
    glBindBuffer(GL_ARRAY_BUFFER, VBO);
    glBufferData(GL_ARRAY_BUFFER, sizeof(verts), verts, GL_STATIC_DRAW);
    // Tell GPU: attribute 0 = 3 floats, starting at offset 0
    glVertexAttribPointer(0,3,GL_FLOAT,GL_FALSE,3*sizeof(float),(void*)0);
    glEnableVertexAttribArray(0);
    glBindVertexArray(0);        // stop recording

    while(!glfwWindowShouldClose(win)){
        if(glfwGetKey(win,GLFW_KEY_ESCAPE)==GLFW_PRESS)
            glfwSetWindowShouldClose(win,true);
        glClearColor(0.1f,0.1f,0.15f,1.0f);
        glClear(GL_COLOR_BUFFER_BIT);
        // Activate shader, bind VAO, draw 3 vertices as a triangle
        glUseProgram(prog);
        glBindVertexArray(VAO);
        glDrawArrays(GL_TRIANGLES, 0, 3);
        glfwSwapBuffers(win);
        glfwPollEvents();
    }
    glDeleteVertexArrays(1,&VAO); glDeleteBuffers(1,&VBO);
    glDeleteProgram(prog);
    glfwTerminate();
}

    ✅ Expected Output

    ✓
    An orange triangle centered on a dark background. The triangle is perfectly centered because NDC (0,0) is screen center, and our vertices span -0.5 to +0.5 in X/Y.

    Rotating 3D Cube with MVP Transforms
    Module 4 · Model-View-Projection · Depth Testing
    Intermediate ›
    ℹ
    What this demo does: Renders a colorful 3D cube that tumbles continuously. Uses the full MVP pipeline — Model (rotates over time), View (camera at position), Projection (perspective). Requires GLM for matrix math.

    🔑 Key Code Snippets

    C++ — MVP Setup (inside render loop)
    // Model: rotate around combined axis — creates tumbling effect
float angle = (float)glfwGetTime() * 50.0f; // 50° per second
glm::mat4 model = glm::mat4(1.0f);
model = glm::rotate(model, glm::radians(angle),
                   glm::vec3(1.0f, 0.5f, 0.3f)); // tumble axis

// View: camera 3 units back, looking at origin
glm::mat4 view = glm::lookAt(
    glm::vec3(0.0f, 1.0f, 3.0f),   // eye position
    glm::vec3(0.0f, 0.0f, 0.0f),   // look at origin
    glm::vec3(0.0f, 1.0f, 0.0f)); // up = Y axis

// Projection: 45° FOV, aspect 800/600, near=0.1, far=100
glm::mat4 proj = glm::perspective(
    glm::radians(45.0f), 800.0f/600.0f, 0.1f, 100.0f);

// Send all three matrices to the vertex shader as uniforms
glUniformMatrix4fv(glGetUniformLocation(prog,"model"),1,GL_FALSE,
                   glm::value_ptr(model));
glUniformMatrix4fv(glGetUniformLocation(prog,"view"), 1,GL_FALSE,
                   glm::value_ptr(view));
glUniformMatrix4fv(glGetUniformLocation(prog,"proj"), 1,GL_FALSE,
                   glm::value_ptr(proj));
    GLSL — Vertex Shader
    #version 330 core
layout(location=0) in vec3 aPos;
layout(location=1) in vec3 aColor;
out vec3 vertColor;
uniform mat4 model, view, proj;
void main(){
    // Matrix chain: proj × view × model × vertex
    gl_Position = proj * view * model * vec4(aPos, 1.0);
    vertColor = aColor;
}

    ✅ Expected Output

    ✓
    A multi-colored cube tumbling in 3D space. Each of the 8 vertices has a unique color (red, green, blue, yellow, magenta, cyan, white, gray). Depth testing ensures front faces always appear in front of back faces.

    Vulkan Instance & Device Initialization
    Module 11 · VkInstance · VkDevice · Queue Families
    Intermediate ›
    ℹ
    What this demo does: The first Vulkan program — no triangle yet. Creates a Vulkan instance with validation layers, selects the best GPU, finds queue families, creates a logical device. Prints all found information. This is the boilerplate all Vulkan apps start with.

    🏗️ Vulkan Object Hierarchy

    VkInstance
    Root. Holds extensions & validation layers
    ↓
    VkPhysicalDevice
    Real GPU — you query, not create
    ↓
    VkDevice
    Logical interface — you create this
    ↓
    VkQueue, Buffers, Pipelines…
    All child objects
    C++ — Vulkan Init (key sections)
    // Step 1: Create Vulkan instance
VkApplicationInfo appInfo{};
appInfo.sType              = VK_STRUCTURE_TYPE_APPLICATION_INFO;
appInfo.pApplicationName   = "RR Graphics Lab";
appInfo.applicationVersion = VK_MAKE_VERSION(1,0,0);
appInfo.apiVersion         = VK_API_VERSION_1_3;

// Always enable validation layers during development!
const std::vector<const char*> layers = {"VK_LAYER_KHRONOS_validation"};

VkInstanceCreateInfo instInfo{};
instInfo.sType                   = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
instInfo.pApplicationInfo        = &appInfo;
instInfo.enabledLayerCount       = (uint32_t)layers.size();
instInfo.ppEnabledLayerNames     = layers.data();
// Add GLFW's required extensions
uint32_t glfwExtCount = 0;
const char** glfwExts = glfwGetRequiredInstanceExtensions(&glfwExtCount);
instInfo.enabledExtensionCount   = glfwExtCount;
instInfo.ppEnabledExtensionNames = glfwExts;

VkInstance instance;
if(vkCreateInstance(&instInfo, nullptr, &instance) != VK_SUCCESS)
    throw std::runtime_error("Failed to create Vulkan instance!");

// Step 2: Pick the best GPU
uint32_t devCount = 0;
vkEnumeratePhysicalDevices(instance, &devCount, nullptr);
std::vector<VkPhysicalDevice> devices(devCount);
vkEnumeratePhysicalDevices(instance, &devCount, devices.data());

VkPhysicalDevice physDev = VK_NULL_HANDLE;
for(auto& dev : devices){
    VkPhysicalDeviceProperties props;
    vkGetPhysicalDeviceProperties(dev, &props);
    std::cout << "GPU found: " << props.deviceName << "\n";
    // Prefer discrete GPU (dedicated card)
    if(props.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU){
        physDev = dev; break;
    }
}
if(physDev == VK_NULL_HANDLE) physDev = devices[0]; // fallback

// Step 3: Find queue families (graphics + present)
uint32_t qfCount = 0;
vkGetPhysicalDeviceQueueFamilyProperties(physDev, &qfCount, nullptr);
std::vector<VkQueueFamilyProperties> qf(qfCount);
vkGetPhysicalDeviceQueueFamilyProperties(physDev, &qfCount, qf.data());

uint32_t graphicsFamily = UINT32_MAX;
for(uint32_t i=0; i<qfCount; i++){
    if(qf[i].queueFlags & VK_QUEUE_GRAPHICS_BIT){
        graphicsFamily = i; break;
    }
}
std::cout << "Graphics queue family: " << graphicsFamily << "\n";
    ⚠
    sType is mandatory: Every Vulkan struct has an sType field. If you forget it, validation layers will flag it immediately. Always set it to the matching VK_STRUCTURE_TYPE_* enum.

    Vulkan First Triangle — Full Pipeline
    Module 12 · Swap Chain · Render Pass · Graphics Pipeline · Sync
    Advanced ›
    ℹ
    What this demo does: The complete Vulkan triangle — swap chain, render pass, full graphics pipeline object, command buffer recording, and semaphore/fence synchronization. ~700 lines total. The most important Vulkan milestone.
    ⚠
    Compile shaders first: Before building, run glslc triangle.vert -o vert.spv and glslc triangle.frag -o frag.spv in the src/ folder. Vulkan requires pre-compiled SPIR-V — it does not accept GLSL source directly.

    🔄 Vulkan Frame Loop

    WaitForFences
    CPU waits for GPU
    →
    AcquireImage
    Get swap image
    →
    Record CmdBuf
    Begin, Draw, End
    →
    QueueSubmit
    Send to GPU
    →
    QueuePresent
    Show on screen
    GLSL — triangle.vert (compile with glslc)
    #version 450
// Hardcoded positions — no vertex buffer needed for first triangle
vec2 positions[3] = vec2[](
    vec2( 0.0, -0.5),  // top
    vec2( 0.5,  0.5),  // bottom-right
    vec2(-0.5,  0.5)   // bottom-left
);
vec3 colors[3] = vec3[](
    vec3(1.0,0.0,0.0), // red
    vec3(0.0,1.0,0.0), // green
    vec3(0.0,0.0,1.0)  // blue
);
layout(location=0) out vec3 fragColor;
void main(){
    gl_Position = vec4(positions[gl_VertexIndex], 0.0, 1.0);
    fragColor    = colors[gl_VertexIndex];
}
    C++ — drawFrame() function
    void drawFrame(){
    // 1. Wait: previous frame must finish before we reuse its command buffer
    vkWaitForFences(device,1,&inFlightFence,VK_TRUE,UINT64_MAX);
    vkResetFences(device,1,&inFlightFence);

    // 2. Acquire next available swap chain image
    uint32_t imgIdx;
    vkAcquireNextImageKHR(device,swapchain,UINT64_MAX,
                          imageAvailSem,VK_NULL_HANDLE,&imgIdx);

    // 3. Record draw commands into command buffer
    vkResetCommandBuffer(cmdBuf,0);
    recordCommandBuffer(cmdBuf, imgIdx);

    // 4. Submit: wait for image, signal when render done
    VkPipelineStageFlags waitStage = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT;
    VkSubmitInfo sub{};
    sub.sType                = VK_STRUCTURE_TYPE_SUBMIT_INFO;
    sub.waitSemaphoreCount   = 1; sub.pWaitSemaphores   = &imageAvailSem;
    sub.pWaitDstStageMask    = &waitStage;
    sub.commandBufferCount   = 1; sub.pCommandBuffers   = &cmdBuf;
    sub.signalSemaphoreCount = 1; sub.pSignalSemaphores = &renderDoneSem;
    vkQueueSubmit(gfxQueue,1,&sub,inFlightFence);

    // 5. Present: show this image once render is done
    VkPresentInfoKHR pres{};
    pres.sType              = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
    pres.waitSemaphoreCount = 1; pres.pWaitSemaphores = &renderDoneSem;
    pres.swapchainCount     = 1; pres.pSwapchains    = &swapchain;
    pres.pImageIndices      = &imgIdx;
    vkQueuePresentKHR(presentQueue,&pres);
}
    🔭
    Sensor Synchronization
    The fence+semaphore pattern ensures sensor data and rendering never overlap — critical for real-time sonar and radar displays.
    ⚡
    Zero Tearing
    Semaphores guarantee the swap image is never displayed mid-render — no tearing on tactical displays even at high update rates.
