Section B — main() function with GLFW setup:

src/main.cpp — Section B: main() Setup
C++
int main() {
    // ── Initialise GLFW ──────────────────────────────────────────────
    if (!glfwInit()) {
        std::cerr << "GLFW init failed\n";
        return -1;
    }

    // Request OpenGL 3.3 Core Profile
    // Must be set BEFORE glfwCreateWindow
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    // ── Create Window + Context ──────────────────────────────────────
    // 1024x768 = common tactical display resolution
    // NULL, NULL = windowed mode, no context sharing
    GLFWwindow* window = glfwCreateWindow(
        1024, 768,
        "Tactical Contact Display — Naval Ops",
        NULL, NULL
    );
    if (!window) {
        std::cerr << "Window creation failed\n";
        glfwTerminate();
        return -1;
    }

    // Make this window's context active on this thread
    // GLEW init will fail if no context is current
    glfwMakeContextCurrent(window);

    // Register resize callback
    glfwSetFramebufferSizeCallback(window, onResize);

    // ── Initialise GLEW ──────────────────────────────────────────────
    // Finds all OpenGL function pointers in the GPU driver
    // Must come after glfwMakeContextCurrent
    glewExperimental = GL_TRUE;
    if (glewInit() != GLEW_OK) {
        std::cerr << "GLEW init failed\n";
        return -1;
    }

    // Print system info — confirms environment is working
    std::cout << "[ Tactical Display System ]\n";
    std::cout << "GPU:            " << glGetString(GL_RENDERER) << "\n";
    std::cout << "OpenGL Version: " << glGetString(GL_VERSION)  << "\n";
    std::cout << "Display ready. Press ESC to shut down.\n";

// aPos comes from our VBO — the x, y, z float triplets we upload
// gl_Position is required — it sets the final clip-space position
const char* vertSrc = R"(

Step 2 — Add Shader Compilation Helper Function

💡 Why a Helper Function?

We compile both shaders the same way (glCreateShader → glShaderSource → glCompileShader → check error). Instead of writing this 8-line block twice, we put it in one function. This is a real pattern used in production OpenGL codebases.

main.cpp — Add above main(): compileShader helper
C++
// Compile a shader from source. Returns its ID.
// type = GL_VERTEX_SHADER or GL_FRAGMENT_SHADER
// source = the GLSL source code string
unsigned int compileShader(unsigned int type, const char* source) {
    unsigned int id = glCreateShader(type);
    glShaderSource(id, 1, &source, NULL);
    glCompileShader(id);

    // Always check for compilation errors
    // The driver compiles GLSL at runtime — syntax errors show here
    int ok; char log[512];
    glGetShaderiv(id, GL_COMPILE_STATUS, &ok);
    if (!ok) {
        glGetShaderInfoLog(id, 512, NULL, log);
        std::cerr << "Shader compile error:\n" << log << "\n";
    }
    return id;
}

// Link vertex + fragment into a complete shader program.
// Returns the program ID.
unsigned int createProgram(const char* vertSrc, const char* fragSrc) {
    unsigned int vs = compileShader(GL_VERTEX_SHADER,   vertSrc);
    unsigned int fs = compileShader(GL_FRAGMENT_SHADER, fragSrc);
    unsigned int prog = glCreateProgram();
    glAttachShader(prog, vs);
    glAttachShader(prog, fs);
    glLinkProgram(prog);
    // Individual shaders no longer needed once linked
    glDeleteShader(vs);
    glDeleteShader(fs);
    return prog;
}

Step 3 — Define Contact Geometry and Upload to GPU

💡 How to Think About This

Three contacts = three small triangles. Each triangle has 3 vertices = 9 floats × 3 contacts = 27 floats total. All 27 floats go into ONE VBO. We draw them all in one glDrawArrays call — start=0, count=9 (9 vertices total = 3 triangles).

main.cpp — inside main(), after glewInit()
C++
    // ── Create Shader Program ────────────────────────────────────────
    unsigned int shaderProg = createProgram(vertSrc, fragSrc);

    // ── Contact Geometry ────────────────────────────────────────────
    // 3 contacts × 3 vertices × 3 floats (x,y,z) = 27 floats
    //
    // Contact 1: FRIENDLY vessel — centre (0.0, 0.6), small triangle
    // Contact 2: HOSTILE vessel  — centre (-0.4, -0.4)
    // Contact 3: UNKNOWN contact — centre (0.5, 0.3)
    //
    // Triangle size: ±0.05 in X, ±0.06 in Y (small tactical icon)
    float contacts[] = {
        // Contact 1 — FRIENDLY (top-centre of display)
        // X      Y      Z
         0.00f, 0.66f, 0.0f,    // tip (top)
        -0.05f, 0.54f, 0.0f,    // bottom-left
         0.05f, 0.54f, 0.0f,    // bottom-right

        // Contact 2 — HOSTILE (lower-left quadrant)
        -0.40f,-0.34f, 0.0f,    // tip
        -0.45f,-0.46f, 0.0f,    // bottom-left
        -0.35f,-0.46f, 0.0f,    // bottom-right

        // Contact 3 — UNKNOWN (right-centre area)
         0.50f, 0.36f, 0.0f,    // tip
         0.45f, 0.24f, 0.0f,    // bottom-left
         0.55f, 0.24f, 0.0f,    // bottom-right
    };

    // ── Upload to GPU with VAO + VBO ─────────────────────────────────
    unsigned int VAO, VBO;
    glGenVertexArrays(1, &VAO);
    glGenBuffers(1, &VBO);

    glBindVertexArray(VAO);              // start recording
      glBindBuffer(GL_ARRAY_BUFFER, VBO);
      glBufferData(GL_ARRAY_BUFFER, sizeof(contacts), contacts, GL_STATIC_DRAW);
      // Tell GPU: attribute 0 = 3 floats, no gap (pure positions only)
      glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3*sizeof(float), (void*)0);
      glEnableVertexAttribArray(0);
    glBindVertexArray(0);               // stop recording

Step 4 — Draw in the Render Loop

main.cpp — replace the render loop body
C++
    while (!glfwWindowShouldClose(window)) {
        processInput(window);

        // Clear to tactical dark background
        glClearColor(0.02f, 0.03f, 0.07f, 1.0f);
        glClear(GL_COLOR_BUFFER_BIT);

        // Activate our shader program
        glUseProgram(shaderProg);

        // Bind VAO — restores VBO + attribute layout
        glBindVertexArray(VAO);

        // Draw all 3 contacts (9 vertices = 3 triangles)
        // GL_TRIANGLES: every 3 vertices = 1 filled triangle
        // 0 = start at vertex index 0
        // 9 = draw 9 vertices total
        glDrawArrays(GL_TRIANGLES, 0, 9);

        glfwSwapBuffers(window);
        glfwPollEvents();
    }

    // Cleanup GPU resources before exit
    glDeleteVertexArrays(1, &VAO);
    glDeleteBuffers(1, &VBO);
    glDeleteProgram(shaderProg);
    glfwTerminate();
    return 0;

Threat Classification Colours
Add per-vertex colour data to the contact markers. Friendly contacts render blue, hostile contacts render red, unknown contacts render yellow. This uses a second vertex attribute (colour) in the same VBO — the interleaved layout pattern.
📋 Concepts: Interleaved VBO · Multiple Attributes · Colour Interpolation ⚡ Repeat first: Demo 4 (VAO stride/offset)
📚 Before You Begin
  • ▸Understand the stride/offset diagram from Demo 4 — draw it on paper if needed
  • ▸Each vertex now has 6 floats: X, Y, Z (position) then R, G, B (colour)
  • ▸The fragment shader needs to receive colour from the vertex shader via out/in
Lab Objectives
  • Update vertex data to include RGB colour per vertex (6 floats per vertex)
  • Update vertex shader to accept and pass colour attribute
  • Update fragment shader to use the interpolated colour
  • Update glVertexAttribPointer for both attributes (correct stride = 24 bytes)
  • Friendly = blue, Hostile = red, Unknown = yellow
Interleaved Vertex Layout — Position + Colour per Vertex
  Each vertex now = 6 floats = 24 bytes:

  Byte: 0    4    8   12   16   20   24   ...
        [X ] [Y ] [Z ] [R ] [G ] [B ] [X ] next vertex...
        ├──position──┤├───colour───┤

  Stride = 24 bytes (6 floats × 4 bytes)
  Position offset = 0  bytes (starts at byte 0)
  Colour   offset = 12 bytes (starts at byte 12, after XYZ)

  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 24, (void*)0 );  ← position
  glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 24, (void*)12);  ← colour
main.cpp — Updated shaders (replace existing)
C++
// Vertex shader now receives TWO attributes:
//   location 0 = position (xyz)
//   location 1 = colour (rgb)
// It passes colour to fragment shader via "out" variable
const char* vertSrc = R"(
    #version 330 core
    layout(location = 0) in vec3 aPos;
    layout(location = 1) in vec3 aColour;

    out vec3 vColour;   // passed to fragment shader (interpolated between vertices)

    void main() {
        gl_Position = vec4(aPos, 1.0);
        vColour = aColour;
    }
)";

// Fragment shader receives interpolated colour from vertex shader
// "in" must match the "out" name in vertex shader exactly
const char* fragSrc = R"(
    #version 330 core
    in vec3 vColour;
    out vec4 FragColor;

    void main() {
        FragColor = vec4(vColour, 1.0);  // RGB from vertex shader, Alpha=1
    }
)";

    • 💡 Design Decision — Separate Draw Calls

    To apply different uniform values to different contacts, we draw each contact separately. In the render loop: draw contact 1 (set uIsHostile=0.0) → draw contact 2 (set uIsHostile=1.0) → draw contact 3 (set uIsHostile=0.0). Each glDrawArrays(... , 0, 3) draws one triangle from offset 0, then 3, then 6 in the VAO.

    main.cpp — Final fragment shader with pulse animation
    C++
    const char* fragSrc = R"(
    #version 330 core

    in vec3 vColour;       // colour passed from vertex shader
    out vec4 FragColor;

    uniform float uTime;       // seconds since program start
    uniform float uIsHostile;  // 1.0 = this contact is hostile, 0.0 = not

    void main() {
        vec3 col = vColour;

        // Apply pulsing effect only to hostile contacts
        // mix(a, b, t) = linear interpolation: a*(1-t) + b*t
        // sin(uTime * 4.0) oscillates fast (4× speed) between -1 and +1
        // (sin() + 1.0) / 2.0 maps to 0.0 - 1.0 for brightness control
        if (uIsHostile > 0.5) {
            float pulse = (sin(uTime * 4.0) + 1.0) / 2.0;
            // At pulse=0.0: dark red (0.3, 0.05, 0.05)
            // At pulse=1.0: bright red (1.0, 0.2, 0.2)
            col = mix(vec3(0.3, 0.05, 0.05), vec3(1.0, 0.2, 0.2), pulse);
        }

        FragColor = vec4(col, 1.0);
    }
)";
    main.cpp — Final render loop with per-contact draw calls
    C++
        // Get uniform locations BEFORE the loop (expensive to call per-frame)
    int uTimeLoc      = glGetUniformLocation(shaderProg, "uTime");
    int uHostileLoc   = glGetUniformLocation(shaderProg, "uIsHostile");

    while (!glfwWindowShouldClose(window)) {
        processInput(window);

        glClearColor(0.02f, 0.03f, 0.07f, 1.0f);
        glClear(GL_COLOR_BUFFER_BIT);

        glUseProgram(shaderProg);

        // Send current time to shader every frame
        float t = (float)glfwGetTime();
        glUniform1f(uTimeLoc, t);

        glBindVertexArray(VAO);

        // ── Contact 1: FRIENDLY — not hostile, no pulse ───────────────
        glUniform1f(uHostileLoc, 0.0f);
        glDrawArrays(GL_TRIANGLES, 0, 3); // vertices 0,1,2

        // ── Contact 2: HOSTILE — pulse animation active ───────────────
        glUniform1f(uHostileLoc, 1.0f);
        glDrawArrays(GL_TRIANGLES, 3, 3); // vertices 3,4,5

        // ── Contact 3: UNKNOWN — not hostile, steady colour ───────────
        glUniform1f(uHostileLoc, 0.0f);
        glDrawArrays(GL_TRIANGLES, 6, 3); // vertices 6,7,8

        // ── Boundary + Grid ───────────────────────────────────────────
        glUniform1f(uHostileLoc, 0.0f);
        glBindVertexArray(gridVAO);
        glDrawElements(GL_LINE_LOOP, 4, GL_UNSIGNED_INT, 0);
        glDrawArrays(GL_LINES, 4, 4);

        glfwSwapBuffers(window);
        glfwPollEvents();
    }
